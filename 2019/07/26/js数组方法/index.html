js数组方法

  1. push 向数组尾部添加新内容，返回数组长度

    var arr1 = [1, 2, 3]
console.log(arr1.push([4, 5]), arr1)

Array.prototype.push1 = function () {
    for (let i = 0; i < arguments.length; i++) {
        this[this.length] = arguments[i];
    }
    return this.length;
}
console.log(arr1.push1([1.2]), arr1)

  2. pop方法是移除数组末尾的元素, 并返回该元素

    var arr2 = []
console.log(arr2.pop(), arr2)

Array.prototype.pop1 = function () {
    var last = this[this.length - 1];
    this.length == 0 ? last = undefined : this.length = this.length - 1
    return last
}
console.log(arr2.pop1(), arr2)

  3. shift会移除数组的第一项元素，并返回该元素

    var arr3 = [1, 2, 3]
console.log(arr3.shift())

Array.prototype.shift1 = function () {
    var first = this[0];
    for (let i = 1; i < this.length; i++) {
        this[i - 1] = this[i]
    }
    this.length = this.length - 1
    return first
}
console.log(arr3.shift1())

  4. unshift方法是向前端添加任意个长度的数组，并返回数组的长度。

    var arr4 = [1, 2, 3, 4]
console.log(arr4.unshift(5, 6, 7), arr4)

Array.prototype.unshift1 = function () {
    var L = this.length;
    for (var i = L + arguments.length - 1; i >= 0; i--) {
        if (i > arguments.length - 1) {
            this[i] = this[i - arguments.length];
        } else {
            this[i] = arguments[i];
        }
    }
    var newarr = []
    for (var i = 0; i < L; i++) {
        newarr[i + arguments.length] = this[i]
        for (var t = 0; t < arguments.length; t++) {
            newarr[t] = arguments[t]
        }
    }
    this == newarr
    return this.length

}
console.log(arr4.unshift1(5, 6, 7), arr4)

  5. concat在没有参数的时候，会复制当前数组，返回副本。但有一个或多个数组，会把这些添加到结果数组中。

    var arr5 = [0, 1, 9, 5]
console.log(arr5.concat('5555, 666'), arr5)

Array.prototype.concat1 = function () {
    var newarr = JSON.parse(JSON.stringify(this))
    if (arguments.length > 0) {
        for (let i = 0; i < arguments.length; i++) {
            newarr[newarr.length] = arguments[i]
        }
        return newarr
    } else {
        return this
    }
}
console.log(arr5.concat1(5555, 666, 7, 8, 9), arr5)

Array.prototype.join1 = function () {
    var str = '',
        ls = '';
    arguments.length == 0 ? ls = ',' : ls = arguments[0]
    for (let i = 0; i < this.length - 1; i++) {
        str += this[i] + "" + ls
    }
    str += this[this.length - 1]
    this.length == 0 ? str = '' : str
    return str
}
